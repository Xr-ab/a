# ===== Day 3：学生成绩管理系统 =====

students = []  # 每个元素 {"name": str, "score": float}

def add_student(name, score):
    students.append({"name": name, "score": score})
    print(f"已添加：{name}，成绩：{score}")

def get_average():
    if not students:
        return 0
    total = sum(s["score"] for s in students)
    return total / len(students)

def get_top_student():
    if not students:
        return None
    return max(students, key=lambda s: s["score"])

def show_all():
    print("\n===== 成绩单 =====")
    for i, s in enumerate(students, 1):
        print(f"{i}. {s['name']}: {s['score']}分")
    print(f"平均分：{get_average():.1f}")
    top = get_top_student()
    if top:
        print(f"最高分：{top['name']}（{top['score']}分）")

- 类与对象 → [廖雪峰](https://www.liaoxuefeng.com/wiki/1016959663602400/1016959739141280) · [菜鸟](https://www.runoob.com/python3/python3-class.html)

**写什么代码**（用类管理成绩）

```python
# ===== Day 5：用类管理成绩 =====
import json

class StudentManager:
    def __init__(self, data_file="students.json"):
        self.data_file = data_file
        self.students = self._load()

    def _load(self):
        try:
            with open(self.data_file, "r", encoding="utf-8") as f:
                return json.load(f)
        except (FileNotFoundError, json.JSONDecodeError):
            return []

    def _save(self):
        with open(self.data_file, "w", encoding="utf-8") as f:
            json.dump(self.students, f, ensure_ascii=False, indent=2)

    def add(self, name, score):
        if not isinstance(score, (int, float)) or score < 0 or score > 100:
            raise ValueError("成绩必须在0-100之间")
        self.students.append({"name": name, "score": score})
        self._save()

    def average(self):
        if not self.students:
            return 0
        return sum(s["score"] for s in self.students) / len(self.students)

    def remove(self, name):
        before = len(self.students)
        self.students = [s for s in self.students if s["name"] != name]
        if len(self.students) == before:
            raise ValueError(f"没有找到学生：{name}")
        self._save()

# 测试
manager = StudentManager()
try:
    manager.add("张三", 85)
    manager.add("李四", 92)
    print(f"平均分：{manager.average():.1f}")
    manager.remove("张三")
    manager.add("王五", 200)  # 会报错
except ValueError as e:
    print(f"出错了：{e}")
```

**检验标准**：能写class，能用try/except处理常见错误

### Day 11 · 多轮对话

**学什么**

- `messages` 列表管理对话历史
- AI怎么"记住"上下文（历史消息每次都发过去）
- `temperature` 参数

📖 **学习资源**

- OpenAI Chat API说明 → [platform.openai.com/docs/guides/chat](https://platform.openai.com/docs/guides/chat)（看"Creating a conversation"那段）
- DeepSeek兼容OpenAI接口，直接用同样的写法

**写什么代码**（命令行聊天）

```python
# ===== Day 11：命令行聊天机器人 =====
import os
from dotenv import load_dotenv
from openai import OpenAI

load_dotenv()
client = OpenAI(
    api_key=os.getenv("DEEPSEEK_API_KEY"),
    base_url="https://api.deepseek.com"
)

messages = [{"role": "system", "content": "你是一个友好的AI助手。"}]

print("AI聊天机器人已启动！（输入 quit 退出）")
while True:
    user_input = input("\n你：")
    if user_input.lower() == "quit":
        print("再见！")
        break

    messages.append({"role": "user", "content": user_input})

    response = client.chat.completions.create(
        model="deepseek-v4-flash"  # 2026.7.24起 deepseek-chat 将废弃，统一用 v4-flash,
        messages=messages,
        temperature=0.7
    )

    reply = response.choices[0].message.content
    print(f"\nAI：{reply}")
    messages.append({"role": "assistant", "content": reply})
```

**检验标准**：连续问3个相关问题，AI能记住上下文

### Day 27 · ⭐ 完整RAG流程（最重要的一天）

把 Day24~26 串起来：

```
文档加载 → 文本切分 → Embedding向量化 → 存入Chroma
                                            ↓
         用户提问 → 检索相似内容 → 喂给AI生成回答
```

📖 **学习资源**

- 吴恩达RAG课程 → [LangChain Chat with Your Data](https://www.deeplearning.ai/short-courses/langchain-chat-with-your-data/)（**1小时免费，RAG必看课**）
- LangChain RAG教程 → [Build a RAG App](https://python.langchain.com/docs/tutorials/rag/)（官方教程，照着走一遍）

```python
# ===== Day 27：完整RAG =====
import os
from dotenv import load_dotenv
from openai import OpenAI
import chromadb
from langchain_community.document_loaders import TextLoader
from langchain_text_splitters import RecursiveCharacterTextSplitter

load_dotenv()

embedding_client = OpenAI(
    api_key=os.getenv("SILICONFLOW_API_KEY"),
    base_url="https://api.siliconflow.cn/v1"
)
chat_client = OpenAI(
    api_key=os.getenv("DEEPSEEK_API_KEY"),
    base_url="https://api.deepseek.com"
)

# 1. 加载文档
loader = TextLoader("你的测试文档.txt", encoding="utf-8")
docs = loader.load()

# 2. 切分
splitter = RecursiveCharacterTextSplitter(chunk_size=500, chunk_overlap=50)
chunks = splitter.split_documents(docs)

# 3. 向量化 + 存入Chroma
chroma = chromadb.PersistentClient(path="./rag_db")
collection = chroma.get_or_create_collection(name="knowledge_base")

for i, chunk in enumerate(chunks):
    emb_response = embedding_client.embeddings.create(
        model="BAAI/bge-large-zh-v1.5",
        input=chunk.page_content
    )
    collection.add(
        documents=[chunk.page_content],
        embeddings=[emb_response.data[0].embedding],
        ids=[f"chunk_{i}"]
    )

print(f"已存入 {len(chunks)} 个文本块到向量数据库")

# 4. 问答循环
while True:
    question = input("\n你的问题（输入quit退出）：")
    if question.lower() == "quit":
        break

    # 检索
    q_emb = embedding_client.embeddings.create(
        model="BAAI/bge-large-zh-v1.5",
        input=question
    )
    results = collection.query(
        query_embeddings=[q_emb.data[0].embedding],
        n_results=3
    )
    context = "\n\n".join(results["documents"][0])

    # 生成回答
    response = chat_client.chat.completions.create(
        model="deepseek-v4-flash"  # 2026.7.24起 deepseek-chat 将废弃，统一用 v4-flash,
        messages=[
            {"role": "system", "content": f"根据以下资料回答问题。资料中没有就说不知道。\n\n参考资料：\n{context}"},
            {"role": "user", "content": question}
        ],
        stream=True
    )

    print("\nAI：", end="")
    for chunk in response:
        if chunk.choices[0].delta.content:
            print(chunk.choices[0].delta.content, end="", flush=True)
    print()
```

**检验标准**：上传文档能基于内容回答问题。**面试必考，必须熟练掌握**

⏱ 4小时（今天多花点时间）

### Day 28 · RAG优化：chunk策略调参

用不同 `chunk_size`（200 / 500 / 1000 / 2000）跑RAG，对比检索质量。

- chunk_size太小 → 信息不完整
- chunk_size太大 → 噪声多，不准
- chunk_overlap → 保留上下文连续性
